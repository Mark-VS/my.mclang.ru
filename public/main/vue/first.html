Vue

Простой пример подключения Vue:

<!-- index.html -->

<html lang="RU">
    ...

  <body>
    <p>Просто бессмысленная строка.<p>

    <div id="app">Говно: {{ message }}</div>

    <script type="module" src="vue_example"></script>
  </body>

</html>

// vue_example.js
    
import { createApp } from "https://unpkg.com/vue@3/dist/vue.esm-browser.js";

const kukuruza = createApp({
   setup() {
       const message = "Строка, которую мы вставим в div.";
       return {
           message
       }
   }
});

kukuruza.mount("#app");

Если всё работает нормально, вместо {{ message }} должна подставиться строка: "Строка, которую мы вставим в div."
