А можно сделать вот так:

server {
   listen   80;
   server_name localhost;

   # Здесь мы вынесли общую для обоих секций папку:
   root   /home/mark/projects/node/my.mclang.ru/public;
   location / {
      # Переопределяем root целиком, чтобы не менять оригинальный URI
      root   /home/mark/projects/node/my.mclang.ru/public/main;
      index index.html index.htm;  # явное указание не обязательно; index.html
      try_files $uri $uri/ =404;
   }
   location /assets/ {
      try_files $uri =404;
   }
   ...
}

Две секции server

Можно указать две секции server с одним и тем же портом. Nginx сначала смотрит порт, а потом - Host header. Если пользователь ввёл IP-адрес вместо домена и ни один домен не совпал, Nginx отдаст первый server-блок.

server {
   listen   80;
   server_name mysite1.ru;

   root /var/www/site1;
   ...
}
server {
   listen   80;
   server_name mysite2.ru;

   root /var/www/site2;
   ...
}

TLS

Если мы хотим дать пользователю возможность заходить через небезопасный http, то:

server {
   listen 80;
   listen [::]:80;

   server_name   test01.mclang.ru;
   root   /home/mark/projects/node/my.mclang.ru/public;
   index  index.html index.htm;
   location / {
      try_files /main$uri /main$uri/index.html /main$uri/index.htm =404;
   }
}
server {
   listen 443 ssl;
   listen [::]:443 ssl;
   ssl_certificate /etc/letsencrypt/live/test01.mclang.ru/fullchain.pem;
   ssl_certificate_key /etc/letsencrypt/live/test01.mclang.ru/privkey.pem;

   server_name   test01.mclang.ru;
   root   /home/mark/projects/node/my.mclang.ru/public;
   index  index.html index.htm;
   location / {
      try_files /main$uri /main$uri/index.html /main$uri/index.htm =404;
   }
}

А если не хотим, то делаем редирект с http на https:

server {
   listen 80;
   listen [::]:80;
   server_name   test01.mclang.ru;

   return 301 https://$host$request_uri;
}
server {
   listen 443 ssl;
   listen [::]:443 ssl;
   ssl_certificate /etc/letsencrypt/live/test01.mclang.ru/fullchain.pem;
   ssl_certificate_key /etc/letsencrypt/live/test01.mclang.ru/privkey.pem;

   server_name   test01.mclang.ru;
   root   /home/mark/projects/node/my.mclang.ru/public;
   index  index.html index.htm;
   location / {
      try_files /main$uri /main$uri/index.html /main$uri/index.htm =404;
   }
}

Если нужно дополнительно обработать адреса с www, то укажем их в server_name:

server {
   listen 80;
   listen [::]:80;
   server_name   test01.mclang.ru www.test01.mclang.ru;

   return 301 https://$host$request_uri;
}
server {
   listen 443 ssl;
