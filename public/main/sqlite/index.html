SQLite

Создать базу данных:

sqlite3 mydb.db

Пример создания таблицы:

-- создаём таблицу заметок
CREATE TABLE notes (
  id INTEGER PRIMARY KEY, -- уникальный ID
  text TEXT               -- текст заметки
);


-- вывести список таблиц:
.tables

-- Полноценный SQL-запрос:
SELECT name FROM sqlite_master WHERE type='table';
