# Запустить раздачу статических файлов на порту 8000:
   python -m http.server

node

# Запустить раздачу статики на порту 3000:
   npx serve
# Запустить раздачу статики на порту 8080 из папки ./public:
   npx serve -l 8080 ./public

# Запустить раздачу статики на порту 8080:
   npx http-server
# Запустить раздачу статики на порту 3000:
   npx http-server ./public -p 3000

# Обновляет страницу при изменении файлов:
   npx lite-server

ruby

# Запустить раздачу статики на порту 8080 из текущей папки:
   ruby -run -e httpd . -p 8080
